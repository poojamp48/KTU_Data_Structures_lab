#include <stdio.h>
#include <stdlib.h>

int main()
{
    int a[20],s ,i,n,f=0,l,mid,found=0;
    printf("Enter the no: of elements:");
    scanf("%d",&n);
    printf("Enter the elements:");
    for(i=0;i<n;i++)
        scanf("%d",&a[i]);
    printf("Enter the element to search for:");
    scanf("%d",&s);
    l=n-1;
    while(f<=l)
    {
        mid=(f+l)/2;
        if(a[mid]==s)
        {
            found=1;
            break;
        }
        else if(a[mid]>s)
        {
            l=mid-1;
        }
        else
            {f=mid+1; }

    }
   if(found==1)
        printf("Element %d found at %d",s,mid+1);
   else
        printf("Element not found");
}
